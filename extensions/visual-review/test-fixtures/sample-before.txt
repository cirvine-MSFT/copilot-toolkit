// Sample configuration for the Widget service
// This file demonstrates a typical before-state for diff testing.

const CONFIG = {
    serviceName: "widget-service",
    version: "1.0.0",
    port: 3000,
    timeout: 5000,
    retries: 3,
};

function initialize(config) {
    console.log("Starting service:", config.serviceName);
    console.log("Port:", config.port);

    if (config.timeout < 1000) {
        throw new Error("Timeout too low");
    }

    return {
        status: "ready",
        uptime: 0,
    };
}

function processRequest(data) {
    if (!data) {
        return { error: "No data provided" };
    }

    const result = {
        id: Math.random().toString(36).slice(2),
        processed: true,
        timestamp: Date.now(),
    };

    return result;
}

function shutdown() {
    console.log("Shutting down gracefully...");
}

module.exports = { CONFIG, initialize, processRequest, shutdown };
